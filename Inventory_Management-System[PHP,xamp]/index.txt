<?php
require 'db_functions.php';
session_start();
//set employee name
if (isset($_GET['user'])) {
    $_SESSION['e_name']= htmlspecialchars($_GET['user']);
}
$e_name=$_SESSION['e_name'];

// Initialize session variables if not already set
if (!isset($_SESSION['items'])) {
    $_SESSION['items'] = [];
}

$subtotal = 0;
$u_name = '';
$phno = '';

// Add new user
if ($_SERVER['REQUEST_METHOD'] == 'POST' && isset($_POST['cus_name'], $_POST['ps'])) {
    $u_name = $_POST['cus_name'];
    $phno = $_POST['ps'];
    
    // Insert new customer into the database
    ins_customer($u_name, $phno);
    
    // Reset user and session items for new user
    $u_name = '';
    $phno = '';
    $_SESSION['items'] = [];
    $subtotal = 0;
    $e_name=$_SESSION['e_name'];
}

// Add item to session 'items' array
if ($_SERVER['REQUEST_METHOD'] == 'POST' && isset($_POST['item_name'], $_POST['item_price'], $_POST['qty'])) {
    $item_name = $_POST['item_name'];
    $item_price = $_POST['item_price'];
    $qty = $_POST['qty'];
    $total = $item_price * $qty;
    
    // Add item data to session
    $_SESSION['items'][] = [
        'item_name' => $item_name,
        'item_price' => $item_price,
        'qty' => $qty,
        'total' => $total
    ];
    $e_name=$_SESSION['e_name'];
}

// Insert session items into the database
if ($_SERVER['REQUEST_METHOD'] == 'POST' && isset($_POST['phone'])) {
    foreach ($_SESSION['items'] as $item) {
        ins_data($_POST['phone'], $item['item_name'], $item['item_price'], $item['qty'], $item['total'], $subtotal);
    }
    
    $e_name=$_SESSION['e_name'];
    // Optionally reset items after insertion
    $_SESSION['items'] = [];
}
// logout
if($_SERVER['REQUEST_METHOD']=='POST' && isset($_POST['logout'])){
    session_unset();
    session_destroy();
    header('Location: logout.html');
    exit();
}
// logout
if($_SERVER['REQUEST_METHOD']=='POST' && isset($_POST['sale_a'])){
    $url = "http://localhost/inventory_system/sale_a.php";

// Output JavaScript to open the page in a new tab when the PHP script is executed
echo "<script type='text/javascript'>
    window.open('$url', '_blank');
</script>";
    $user = urlencode($e_name);
    header("Location: sale_a.php?user=".$user);
    header('Location: sale_a.php');
}

}
